```
tccli dbbrain DescribeSlowLogTopSqls --cli-unfold-argument  \
    --InstanceId cdb-test1234 \
    --SortBy QueryTime \
    --OrderBy ASC \
    --Limit 10 \
    --Offset 0 \
    --StartTime '2019-01-01 00:00:00' \
    --EndTime '2019-01-01 01:00:00' \
    --SchemaList.0.Schema dbName
```
