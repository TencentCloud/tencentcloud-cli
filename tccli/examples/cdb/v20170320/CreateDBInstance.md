```

    --ResourceTags.0.TagKey march \
    --ResourceTags.0.TagValue march1 \
    --Zone ap-guangzhou-3 \
    --UniqVpcId vpc-0akbol5v \
    --ProjectId 0 \
    --InstanceRole master \
    --GoodsNum 1 \

    --Volume 25 \
```
