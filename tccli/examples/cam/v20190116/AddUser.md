```
tccli cam AddUser --cli-unfold-argument  \
    --Remark jack account \
    --Name jack \
    --CountryCode 86 \
    --NeedResetPassword 0 \
    --PhoneNum 10086 \
    --UseApi 1 \
    --ConsoleLogin 1 \
    --Password test123456 \
    --Email 123%40qq.com
```
